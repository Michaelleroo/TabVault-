const supabase = window.supabase.createClient(SUPABASE_URL, SUPABASE_ANON); async function trackEvent(eventName, details = '') { try { const { data: { user } } = await supabase.auth.getUser(); if (!user) return; await supabase.from('events').insert({ user_id: user.id, event_name: eventName, details: details }); } catch (err) { console.error('Tracking error:', err); } }
